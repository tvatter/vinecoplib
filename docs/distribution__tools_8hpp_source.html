107 inline Eigen::MatrixXd simulate_uniform(int n, int d)
108 {
109  if ((n < 1) | (d < 1))
110  throw std::runtime_error("both n and d must be at least 1.");
111  Eigen::MatrixXd U(n, d);
112  std::random_device rd;
113  std::default_random_engine generator(rd());
114  std::uniform_real_distribution<double> distribution(0.0, 1.0);
115  return U.unaryExpr([&](double) { return distribution(generator); });
116 }
